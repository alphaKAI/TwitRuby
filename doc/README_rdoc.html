    Usage¶ ↑

    Please require this liblary.

    #code
#define
consumer_key = ""
consumer_secret = ""
access_token = ""
access_token_secret = ""
#Authentication
cunsmer_array=[]
cunsmer_array << consumer_key << consumer_secret << access_token << access_token_secret
twi=TwitRuby.new
twi.initalize_connection(cunsmer_array)
#example
twi.update("sample")#example, update tweet.
